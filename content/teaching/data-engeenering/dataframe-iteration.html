def df_itertuples(df):
    for t in df.itertuples():
        _ = t.one  # simple element access


def df_iterrows(df):
    for i, row in df.iterrows():
        _ = row.one  # simple element access


def df_apply(df):
    df.apply(
        lambda x: x.one, axis=1  # simple element access
    )
